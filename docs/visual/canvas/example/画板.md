```js
// 1.获取canvas画布和绘制的上下文对象
var canvas = document.getElementById('c1')
var context = canvas.getContext('2d')
// 2. 获取输入框和按钮
let boldBtn = document.querySelector('#boldBtn')
let thinBtn = document.querySelector('#thinBtn')
let saveBtn = document.querySelector('#saveBtn')
let inputColor = document.querySelector('#inputColor')
let clearBtn = document.querySelector('#clearBtn')
let nullBtn = document.querySelector('#nullBtn')

// 连接初圆润
context.lineJoin = 'round'

// 设置允许绘制的变量
var isDraw = false
canvas.onmousedown = function () {
  isDraw = true
  context.beginPath()
  var x = event.pageX - canvas.offsetLeft
  var y = event.pageY - canvas.offsetTop
  context.moveTo(x, y)
}
canvas.onmouseleave = function () {
  isDraw = false
  context.closePath()
}
canvas.onmouseup = function () {
  isDraw = false
  context.closePath()
}
canvas.onmousemove = function () {
  if (isDraw) {
    var x = event.pageX - canvas.offsetLeft
    var y = event.pageY - canvas.offsetTop
    context.lineTo(x, y)
    context.stroke()
  }
}

// 粗线条
boldBtn.onclick = function () {
  context.globalCompositeOperation = 'source-over'
  context.lineWidth += 5
}
// 细线条
thinBtn.onclick = function () {
  context.globalCompositeOperation = 'source-over'
  if (context.lineWidth <= 0) {
    context.lineWidth = 1
  } else {
    context.lineWidth -= 5
  }
}

// 保存签名
saveBtn.onclick = function () {
  var urlData = canvas.toDataURL()
  // var img = new Image();
  // img.src = urlData;
  // document.body.appendChild(img);

  var downloadA = document.createElement('a')
  downloadA.setAttribute('download', '签名')
  downloadA.href = urlData
  downloadA.click()
}

// 改变颜色
inputColor.onchange = function () {
  context.strokeStyle = inputColor.value
}

// 橡皮擦
clearBtn.onclick = function () {
  context.globalCompositeOperation = 'destination-out'
}

// 清空画布
nullBtn.onclick = function () {
  context.clearRect(0, 0, 1000, 600)
}
```
